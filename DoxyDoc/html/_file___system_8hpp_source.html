24/**
25 * @brief Helper function to print std::array<std::byte, N> to std::ostream
26 * @tparam N Size of the array
27 * @param os Output stream
28 * @param v1 Array to print
29 * @return Output stream
30 */
31template<std::size_t N>
32std::ostream &operator<<(std::ostream &os, std::array<std::byte, N> const &v1) {
33
34 std::for_each(v1.begin(), v1.end(), [&os](std::byte b) {
35 os << std::hex << static_cast<int>(b) << " ";
36 });
37
38 return os;
39}

201 /**
202 * @brief Structure for the SuperBlock
203 * @details The SuperBlock is the first block of the disk and contains information about the file system
204 * @details The SuperBlock is 4KB in size
205 */
206 struct SuperBlock {
207 SuperBlock() : MagicNumber(0), Blocks(0), InodeBlocks(0), DirBlocks(0), Inodes(0), PasswordHash(), Protected(0) {}
208
209 u_int32_t MagicNumber; // Magic Number to identify the file system
210 u_int32_t Blocks; // Number of blocks in the file system
211 u_int32_t InodeBlocks; // Number of Inode blocks in the file system
212 u_int32_t Inodes; // Number of Inodes in the file system
213 u_int32_t DirBlocks; // Number of Directory blocks in the file system
214
215 std::array<char, 257> PasswordHash; // Hash of the password
216 u_int32_t Protected; // 1 if the file system is protected; 0 otherwise
217 };

219 /**
220 * @brief Structure for the Directory Entry
221 * @details The Directory Entry contains information about a file or directory
222 * @details The Directory Entry is 32 bytes in size
223 */
224 struct Dirent {
225 uint8_t type = 0; // 0 if the entry is a file; 1 if the entry is a directory
226 uint8_t valid = 0; // 1 if the entry is valid; 0 otherwise
227 uint32_t i_num = 0; // The Inode number of the file or directory
228
229 std::array<char, NAME_SIZE> name{}; // The name of the file or directory
230 };

264 /**
265 * @brief Structure for the Inode
266 * @details The Inode contains information about a file
267 * @details The Inode is 128 bytes in size
268 */
269 struct Inode {
270 uint32_t Valid; // 1 if the Inode is valid; 0 otherwise
271 uint32_t Size; // The size of the file in bytes
272 std::array<uint32_t, FileSystem::POINTERS_PER_INODE> Direct; // The direct pointers to the data blocks
273 uint32_t Indirect; // The indirect pointer to the data blocks
274
275 /**
276 * @brief Default constructor
277 */
278 Inode() : Valid(0), Size(0), Direct(), Indirect(0) {}
279 };

284 * @details The Block is 4KB in size
285 */
286 union Block {
287
288 SuperBlock Super; // The SuperBlock
289 std::array<Inode, FileSystem::INODES_PER_BLOCK> Inodes; // The Inodes
290 std::array<std::uint32_t, FileSystem::POINTERS_PER_BLOCK> Pointers; // The pointers to the data blocks
291 std::array<std::byte, Disk::BLOCK_SIZE> Data; // The data in the block
292 std::array<Directory, FileSystem::DIR_PER_BLOCK> Directories; // The directories
293
294 /**
295 * @brief Default constructor
296 */
297 Block() : Data() {}

299 };
